import java.util.*;

public class Main {

	public static void main(String[] args) {
		//fill your code here
		Scanner scanner = new Scanner(System.in);
		
		int num, rev = 0;

		System.out.println("Enter the number :");
		num = scanner.nextInt();

		while(num > 0) {
			rev = rev*10 + num%10;
			num /= 10;
		}

		System.out.println("Reverse of the number is " + rev);
	}

}
